#include<iostream>
using namespace std;
typedef int ElemType;
#define MAXSIZE 10
//定义LNode结构的指针类型为List
typedef struct LNode *List;

//以下为LNode结构的声明
struct LNode
{
    ElemType data[MAXSIZE];
    int Last;//最大位置的index

};
struct LNode L;//生成一个LNode的结构
List Ptrl;//指向LNode结构的指针

//1建立空的顺序表
List MakeEmpty()
{
    List Ptrl;
    Ptrl = (List)malloc(sizeof(struct LNode));
    Ptrl->Last = -1;//没元素就设为-1,有一个的话就是0了
    return Ptrl;
}

//2.查找List表中的元素X
int Find(ElemType x, List Ptrl)
{
    int i = 0;
    while (i <= Ptrl->Last&& Ptrl->data[i] != x)
        i++;
    if (i > Ptrl->Last) return -1;
    else return i;
}

//3.把x插入到L线性表的第i个位置
void Insert(ElemType x, int i, List Ptrl)
{
    int j;
    //判断空间是否满了
    if (Ptrl->Last == MAXSIZE - 1)
    {
        printf("表满了");
        return;
    }
    //检查插入位置的合法性, 是第i个位置,所以不能为0或Last+2
    if (i<1 || i>Ptrl->Last + 2)
    {
        printf("插入位置不合法");
        return;
    }

    //插入
    for (j = Ptrl->Last + 1; j > i-1; j--)
    {
        Ptrl->data[j] = Ptrl->data[j - 1];
    }
    Ptrl->data[i - 1] = x;//插入x
    Ptrl->Last++; //不要忘了Last
    return;
}

//4.删除L表的第i个元素
void Delet(int i, List L)
{
    //判断
    if (i<1 || i>L->Last + 1)
        printf("删除的位置不合法");

        for (int j = i - 1; j < L->Last; j++)
        {
            L->data[j] = L->data[j + 1];

        }
        L->Last--;
}

int main()
{
    List Ptrl = MakeEmpty();
    //插入6个数
    for (ElemType i = 1; i < 11; i++)
    {

        Insert(i, i, Ptrl);
        cout << "此时的线性表为";
        for (int j = 0; j <= Ptrl->Last; j++)
            cout << Ptrl->data[j];
        cout << endl;
    }

    //使用函数2查找元素3的下标
    int find_result=Find(3, Ptrl);

    //插入到L1第6个位置
    Insert(2, 6, Ptrl);

    cout << "此时的线性表为";
    for (int j = 0; j <=Ptrl->Last; j++)
        cout << Ptrl->data[j];
    cout << endl;
    //删除L表的第6个位置
    Delet(6, Ptrl);
    cout << "此时的线性表为";
    for (int j = 0; j <= Ptrl->Last; j++)
        cout << Ptrl->data[j];
    cout << endl;

}
